int factorial(int n) {
  if (n < 0) {
    throw ArgumentError("n phải dương;-;");
  }
  int result = 1;
  for (int i = 2; i <= n; i++) {
    result *= i;
  }
  return result;
}

void main() {
  int num = 5;
try {
    print("Factorial of $num is ${factorial(num)}"); 
  } catch (e) {
    print("Error: $e");
  }
}
